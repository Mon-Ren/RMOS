汇编级优化通过消除分支、利用 SIMD、减少内存访问等技巧榨取 CPU 的最后一点性能，在高性能计算和系统编程中至关重要。

## 1. 消除分支：CMOV 和 SETcc

分支是性能的大敌（分支预测失败 ~15 周期惩罚）。用条件传送代替条件跳转：

```nasm
; ❌ 有分支的写法
    cmp     eax, ebx
    jge     .skip
    mov     ecx, eax        ; ecx = min(a, b)
.skip:

; ✅ 无分支 — CMOVcc
    cmp     eax, ebx
    mov     ecx, eax
    cmovl   ecx, ebx        ; if (a < b) ecx = ebx

; ✅ 无分支 — SETcc（布尔结果）
    cmp     eax, ebx
    setl    cl              ; cl = (a < b) ? 1 : 0
    movzx   ecx, cl
```

### C 语言编译器优化

```c
// 写法 1: 分支（可能被优化为 CMOV）
int max_branch(int a, int b) {
    return (a > b) ? a : b;
}

// 写法 2: 无分支位运算
int max_bitwise(int a, int b) {
    int mask = ~(a - b) >> 31;   // a >= b → 0xFFFFFFFF, 否则 → 0
    return (a & mask) | (b & ~mask);
}

// 写法 3: 技巧 — 用 sign bit 做选择
int abs_branchless(int x) {
    int mask = x >> 31;           // 负数 → 0xFFFFFFFF
    return (x ^ mask) - mask;     // (x ^ -1) - (-1) = ~x + 1 = -x
}
```

## 2. 循环展开

```nasm
; ❌ 未展开的循环
.loop:
    add     eax, [esi]
    add     esi, 4
    dec     ecx
    jnz     .loop

; ✅ 4 次展开
.loop4:
    add     eax, [esi]
    add     eax, [esi+4]
    add     eax, [esi+8]
    add     eax, [esi+12]
    add     esi, 16
    sub     ecx, 4
    jnz     .loop4

; ✅ 带 SIMD 的展开（更好）
.loop_simd:
    movdqu  xmm0, [esi]          ; 加载 4 个 int
    paddd   xmm2, xmm0           ; 累加
    add     esi, 16
    sub     ecx, 4
    jnz     .loop_simd
```

**展开的好处**：
- 减少循环开销（DEC + JNZ 有 ~1-2 周期）
- 增加指令级并行
- 让编译器更好地调度指令

**注意**：现代 CPU 有循环缓冲区和分支预测，过度展开可能适得其反。

## 3. 利用 SIMD 并行

```nasm
; 标量版本：4 条 ADD
add eax, [esi]
add eax, [esi+4]
add eax, [esi+8]
add eax, [esi+12]

; SIMD 版本：1 条 ADDPS
movups  xmm0, [esi]              ; 加载 4 个 float
addps   xmm0, xmm1               ; 4 个加法并行
```

```c
// 编译器自动向量化提示
#pragma GCC optimize("O3")
void scale_array(float *a, float k, int n) {
    // 编译器可能自动使用 SIMD
    for (int i = 0; i < n; i++)
        a[i] *= k;
    
    // 或者手动使用 SIMD 内联函数
    __m256 vk = _mm256_set1_ps(k);
    for (int i = 0; i < n; i += 8) {
        __m256 va = _mm256_load_ps(&a[i]);
        _mm256_store_ps(&a[i], _mm256_mul_ps(va, vk));
    }
}
```

## 4. 减少内存访问：寄存器分配

```nasm
; ❌ 频繁访问内存
.loop:
    mov     eax, [counter]
    add     eax, 1
    mov     [counter], eax        ; 每次循环都写内存
    dec     ecx
    jnz     .loop

; ✅ 在寄存器中累加
    mov     eax, [counter]         ; 只读一次
.loop:
    inc     eax                     ; 寄存器操作
    dec     ecx
    jnz     .loop
    mov     [counter], eax          ; 只写一次
```

**规则**：
- 热循环中的变量尽量放寄存器
- 避免循环内的内存读写
- 如果必须写内存，考虑批量写

## 5. 避免假依赖：XOR 清零

```nasm
; ❌ 假依赖 — MOV EAX, 0 需要等待旧 EAX
    mov     eax, [val]
    ; ... 一些操作 ...
    mov     eax, 0              ; 假依赖：CPU 可能认为需要旧值

; ✅ XOR 清零 — 打破依赖链
    xor     eax, eax            ; CPU 识别为清零，无依赖

; ✅ SUB 自清零
    sub     eax, eax            ; 同上

; ✅ LEA 计算（不修改标志位）
    lea     eax, [0]            ; eax = 0, 不影响 EFLAGS
```

## 6. 内存对齐

```nasm
; 对齐的数据访问更快
section .data
    align 16
    data: dd 1.0, 2.0, 3.0, 4.0   ; 16 字节对齐

section .text
    movaps  xmm0, [data]            ; 对齐加载 — 单周期
    ; movups xmm0, [data]           ; 非对齐 — 可能需要两次访问

; 循环中保证对齐
    test    esi, 0x0F               ; 检查是否 16 字节对齐
    jnz     .unaligned_path
    movaps  xmm0, [esi]             ; 快速路径
    jmp     .continue
.unaligned_path:
    movups  xmm0, [esi]             ; 慢速路径
.continue:
```

## 7. 数据预取

```nasm
; 大数组处理时预取后续数据
.loop:
    prefetchnta [esi+64]             ; 预取 64 字节后的数据
    ; 处理当前数据
    movdqa  xmm0, [esi]
    ; ...
    add     esi, 16
    dec     ecx
    jnz     .loop
```

## 完整优化示例：字符串长度

```nasm
; size_t strlen(const char *s)
; 基础版本
strlen_basic:
    xor     eax, eax
.loop:
    cmp     byte [rdi + rax], 0
    je      .done
    inc     rax
    jmp     .loop
.done:
    ret

; 优化版本 — 使用 SIMD 检查 16 字节
strlen_simd:
    mov     rsi, rdi
    pxor    xmm0, xmm0              ; xmm0 = 0（用于比较）
    
    ; 对齐到 16 字节
    and     rdi, -16
    and     esi, 15                 ; 偏移量
    pcmpeqb xmm1, [rdi]             ; 比较 16 字节
    pmovmskb eax, xmm1              ; 提取比较结果
    shr     eax, cl                 ; 移除前导无效位
    test    eax, eax
    jnz     .found
    
.loop16:
    add     rdi, 16
    pcmpeqb xmm0, [rdi]
    pmovmskb eax, xmm0
    test    eax, eax
    jz      .loop16

.found:
    bsf     eax, eax                ; 找到第一个为 1 的位
    add     rax, rdi
    sub     rax, rsi
    ret
```

## 性能测量

```c
// 使用 RDTSC 测量周期数
static inline uint64_t rdtsc(void) {
    uint32_t lo, hi;
    __asm__ __volatile__ ("rdtsc" : "=a"(lo), "=d"(hi));
    return ((uint64_t)hi << 32) | lo;
}

void benchmark(void) {
    // 预热缓存
    for (int i = 0; i < 1000; i++) my_function();
    
    uint64_t start = rdtsc();
    for (int i = 0; i < 100000; i++)
        my_function();
    uint64_t end = rdtsc();
    
    printf("Avg cycles: %.1f\n", (end - start) / 100000.0);
}
```

## 关键要点

> **CMOV > 条件跳转**：无分支 = 无预测失败惩罚。简单的 if-else 优先用 CMOV。

> **SIMD 是最大的加速来源**：4/8/16 倍数据并行，配合循环展开效果最佳。

> **寄存器 > 内存**：热循环中的变量必须在寄存器中。频繁读写内存是最大的性能浪费。

> **XOR reg, reg 是标准清零方式**：编译器和手写汇编都用这个，因为它打破依赖链，且编码更短。

> **对齐很重要**：`MOVAPS`（对齐）比 `MOVUPS`（非对齐）在老 CPU 上快一倍。
