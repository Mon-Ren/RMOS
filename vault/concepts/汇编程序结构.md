汇编程序由多个**段（section/segment）**组成，每段有明确的职责——代码、已初始化数据、未初始化数据，这是 ELF 可执行文件组织的基础。

## 三大基本段

```nasm
; 完整的汇编程序结构
section .text           ; 代码段 — 可执行指令
global _start

_start:
    ; 你的代码
    mov eax, 1          ; sys_exit
    xor ebx, ebx        ; return 0
    int 0x80

section .data           ; 已初始化数据段 — 有初始值的全局/静态变量
    msg     db "Hello", 10
    msg_len equ $ - msg ; $ = 当前地址, $-msg = 字符串长度
    num     dd 42
    table   times 256 db 0  ; 256 字节的零填充

section .bss            ; 未初始化数据段 — 预留空间，不占文件体积
    buffer  resb 1024   ; 预留 1024 字节
    stack   resb 4096   ; 预留 4096 字节的栈空间
    array   resd 100    ; 预留 100 个双字
```

### 段的属性

| 段 | ELF 属性 | 内存权限 | 文件大小 | 内存大小 |
|----|----------|---------|---------|---------|
| `.text` | `PROGBITS, AX` | r-x | = 实际大小 | = 实际大小 |
| `.data` | `PROGBITS, WA` | rw- | = 实际大小 | = 实际大小 |
| `.bss` | `NOBITS, WA` | rw- | 0 | = 预留大小 |

> `.bss` 在文件中不占空间——加载器只分配内存并清零。

## 符号定义与引用

```nasm
; 定义全局符号（其他文件可引用）
global my_function
global my_variable
global _start

; 引用外部符号（在链接时解析）
extern printf
extern malloc
extern errno

; 弱符号（可以被覆盖）
global my_default_handler:default    ; NASM 语法

; 局部符号（不导出，名称以 . 开头）
.local_label:
    jmp .local_label
```

### 符号可见性

```nasm
; NASM
global func_name              ; 默认全局
global func_name:default      ; 弱符号

; GAS
.globl func_name              ; 全局
.weak func_name               ; 弱符号
.local  local_var             ; 局部符号
.hidden hidden_func           ; 隐藏符号
```

## 入口点：_start vs main

```nasm
; === _start（底层入口）===
; 由链接器直接指定，不需要 C 运行时
global _start
_start:
    ; 此时没有栈参数解析
    ; ESP 指向 argc（从内核接收）
    pop  ecx              ; argc
    mov  esi, esp         ; argv[]
    ; ...
    mov  eax, 1           ; sys_exit
    int  0x80

; === main（C 运行时入口）===
; 由 crt0.o 调用，已解析好 argc/argv/envp
global main
main:
    push ebp
    mov  ebp, esp
    ; 参数: [ebp+8]=argc, [ebp+12]=argv
    ; ...
    pop  ebp
    ret
```

**区别**：

| | `_start` | `main` |
|---|----------|--------|
| 调用者 | 链接器（直接跳转） | C 运行时（crt0.o） |
| 前置处理 | 无 | 初始化堆栈、调用 atexit |
| 系统调用 | 直接 `int 0x80` / `syscall` | 可用 C 库函数 |
| 退出 | 直接 sys_exit | `return` 或 `exit()` |

## 其他常用段

```nasm
section .rodata           ; 只读数据
    const_str  db "Read-only string", 0
    const_val  dd 0xDEADBEEF

section .init             ; 初始化代码（main 之前执行）
    call setup

section .fini             ; 清理代码（main 之后执行）
    call cleanup

; 自定义段
section .mydata progbits alloc noexec write align=16
    custom_data db 16 dup(0)
```

## 链接脚本简介

链接脚本控制最终可执行文件的段布局：

```ld
/* 最小链接脚本示例 */
ENTRY(_start)              /* 入口点 */

SECTIONS {
    . = 0x08048000;        /* 加载地址（32 位常用） */
    
    .text : {
        *(.text)           /* 所有 .text 段合并 */
    }
    
    .rodata : {
        *(.rodata)
    }
    
    .data : {
        *(.data)
    }
    
    .bss : {
        *(.bss)
        *(COMMON)          /* 未初始化的全局变量 */
    }
}
```

```bash
# 使用自定义链接脚本
ld -T myscript.ld -o program obj1.o obj2.o
```

## 完整示例：从源码到可执行

```nasm
; hello.asm — 完整的汇编程序
section .data
    msg db "Hello, World!", 10
    len equ $ - msg

section .text
    global _start

_start:
    ; write(1, msg, len)
    mov eax, 4          ; sys_write
    mov ebx, 1          ; fd = stdout
    mov ecx, msg        ; buf
    mov edx, len        ; count
    int 0x80

    ; exit(0)
    mov eax, 1          ; sys_exit
    xor ebx, ebx        ; status = 0
    int 0x80
```

```bash
# 编译和链接
nasm -f elf32 hello.asm -o hello.o    # 汇编
ld -m elf_i386 hello.o -o hello        # 链接
./hello                                 # 运行
```

## 关键要点

> **三大段**：`.text`(代码, r-x), `.data`(已初始化数据, rw-), `.bss`(未初始化数据, rw-)。`.bss` 不占磁盘空间，加载时清零。

> **_start 是真正的入口**：不依赖 C 运行时。`main` 需要 `gcc` 链接 `crt0.o`。裸机/内核开发必须用 `_start`。

> **符号**：`global` 导出, `extern` 引用外部。符号在链接时统一解析。

> **链接脚本**控制最终布局——对内核和 bootloader 开发至关重要。
