CPU 缓存是处理器与主存之间的高速缓冲层，理解缓存层次和缓存行（Cache Line）对编写高性能代码至关重要。

## 缓存层次结构

```
┌─────────────┐
│  CPU Core   │
│  ┌───────┐  │   ┌────────────────┐
│  │ L1-I  │  │   │ L1-I: 指令缓存 │ ~32KB, 4路, ~4周期
│  └───────┘  │   │ L1-D: 数据缓存 │ ~32KB, 8路, ~4周期
│  ┌───────┐  │   └────────────────┘
│  │ L1-D  │  │
│  └───────┘  │   ┌────────────────┐
│  ┌───────┐  │   │ L2: 统一缓存   │ ~256KB-1MB, ~12周期
│  │  L2   │  │   └────────────────┘
│  └───────┘  │
└─────────────┘
┌───────────────────────────────────┐
│              L3 共享缓存           │  ~4-64MB, ~40周期
└───────────────────────────────────┘
┌───────────────────────────────────┐
│            主内存 (DRAM)           │  ~GB 级, ~200 周期
└───────────────────────────────────┘
```

## 缓存行（Cache Line）

```
缓存行大小：通常 64 字节（x86）

每次缓存未命中时，不是只读一个字节，而是加载整个 64 字节的块：

内存地址 0x1000 - 0x103F 全部加载到一条缓存行中

这 64 字节 = 连续 16 个 int 或 8 个 double
```

### 缓存行对齐的重要性

```c
// 好：结构体大小是缓存行的倍数
struct __attribute__((aligned(64))) cache_aligned {
    int data[16];    // 正好 64 字节，填满一个缓存行
};

// 坏：跨缓存行访问
struct bad_layout {
    char a;          // 偏移 0
    // 63 字节填充...
    char b;          // 偏移 64 — 在另一个缓存行！
};

// 更好：用 aligned attribute
struct counter {
    int value;
} __attribute__((aligned(64)));  // 每个 counter 独占一个缓存行
```

## 缓存命中与未命中

```c
// === 缓存友好的访问模式 ===
// 行优先遍历（缓存命中率高）
for (int i = 0; i < N; i++)
    for (int j = 0; j < N; j++)
        sum += matrix[i][j];   // 连续内存访问

// === 缓存不友好的访问模式 ===
// 列优先遍历（大量缓存未命中）
for (int j = 0; j < N; j++)
    for (int i = 0; i < N; i++)
        sum += matrix[i][j];   // 每次跳 N*4 字节，每次都是未命中
```

## 伪共享（False Sharing）

```
核心 A 写 counter_a（在缓存行 #42 的开头）
核心 B 写 counter_b（在缓存行 #42 的结尾）

虽然它们操作不同变量，但共享同一条缓存行！
每次写入都导致另一核心的缓存行失效 → 频繁的缓存一致性流量
```

```c
// ❌ 伪共享示例
struct {
    int core_a_counter;    // 偏移 0
    int core_b_counter;    // 偏移 4 — 同一缓存行！
} shared;

// ✅ 消除伪共享
struct {
    int core_a_counter;
    char padding[60];      // 填充到 64 字节
} __attribute__((aligned(64)));

struct {
    char padding[64];      // 跳过一个缓存行
    int core_b_counter;
} __attribute__((aligned(64)));

// 或者用 C11 alignas
#include <stdalign.h>
struct {
    alignas(64) int core_a_counter;
    alignas(64) int core_b_counter;
};
```

### 性能影响

```bash
# 伪共享的性能差异可达 10 倍以上
# 使用 perf 检测
perf stat -e cache-misses,cache-references ./program
```

## 缓存关联性

```
直接映射：每个内存地址只能映射到一条缓存行 — 冲突多
全相联：  任何地址可以放在任意位置 — 查找慢
组相联：  N 路组相联（常见 8 路）— 折中方案

x86 典型配置：
  L1-D: 32KB, 8路组相联, 64B 行 → 64 组
  L2:   256KB, 8路组相联 → 512 组
```

## 汇编级缓存控制

```nasm
; === CLFLUSH — 刷新缓存行 ===
clflush [addr]             ; 将包含 addr 的缓存行写回并无效化
clflushopt [addr]          ; 优化版本（无序）

; === CLWB — 写回但保留（SKX+）===
clwb [addr]                ; 写回缓存行，但保持在缓存中

; === 非时间存储 — 绕过缓存 ===
movntdq [addr], xmm0       ; 128 位非时间存储（不污染缓存）
movntps [addr], xmm0       ; float 非时间存储
movnti  [addr], eax        ; 32 位整数非时间存储

; === PREFETCH — 预取 ===
prefetcht0 [addr]          ; 预取到所有缓存层
prefetcht1 [addr]          ; 预取到 L2 和更上层
prefetcht2 [addr]          ; 预取到 L3
prefetchnta [addr]         ; 预取到 L1（非时间，低优先级）
```

### 缓存友好的 SIMD 循环

```nasm
; 大数组处理 — 使用非时间存储避免缓存污染
process_large_array:
    ; 每次处理 128 字节（2 条缓存行）
    movdqa  xmm0, [rsi]
    ; ... 处理 ...
    movntdq [rdi], xmm0    ; 非时间存储 — 不污染缓存

    add     rsi, 16
    add     rdi, 16
    dec     rcx
    jnz     process_large_array

    sfence                  ; 确保非时间存储完成
```

## 缓存友好的数据结构

```c
// ✅ SoA (Structure of Arrays) — SIMD 友好
struct particles {
    float x[1024];    // 所有 x 连续存储
    float y[1024];    // 所有 y 连续存储
    float z[1024];
};
// 遍历时，x[i] 全在缓存中

// ❌ AoS (Array of Structures)
struct particle {
    float x, y, z;   // 12 字节，跨缓存行
};
struct particle particles[1024];
// 遍历 x 时，y 和 z 浪费缓存空间
```

## 关键要点

> **缓存行 = 64 字节**：这是 CPU 读取内存的最小粒度。跨缓存行访问需要两次内存事务。

> **伪共享是多线程性能杀手**：不同核心写同一缓存行的不同位置，导致缓存行反复失效。用 padding 隔离。

> **访问模式至关重要**：顺序访问（行优先）远快于跳跃访问（列优先），因为前者充分利用预取和缓存行。

> **非时间存储**：大块数据的流式处理用 `MOVNTDQ` 避免污染缓存。
