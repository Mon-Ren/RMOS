原子操作指令是多处理器同步的硬件基础，x86 通过 `LOCK` 前缀和专用原子指令实现不可中断的读-改-写操作。

## XCHG — 天生原子

```nasm
; XCHG 隐含 LOCK（无论是否显式加前缀）
xchg    eax, [lock_var]     ; 原子交换：EAX ↔ [lock_var]

; 自旋锁实现
acquire_lock:
    mov     eax, 1
    xchg    eax, [lock]     ; 原子写入 1，返回旧值
    test    eax, eax        ; 旧值 == 0？
    jnz     acquire_lock    ; 不为 0，继续自旋
    ret                     ; 获得锁

release_lock:
    mov     dword [lock], 0 ; 释放锁（x86 保证对齐的 32 位写是原子的）
    ret
```

> **关键**：`XCHG` 是唯一不需要 `LOCK` 前缀就天生原子的 x86 指令。

## LOCK 前缀 — 总线锁/缓存锁

```nasm
; LOCK 前缀使后续指令变为原子操作

; 原子加
lock add    dword [counter], 1

; 原子减
lock sub    dword [counter], 1

; 原子或/与/异或
lock or     dword [flags], 0x01
lock and    dword [flags], 0xFFFFFFFE
lock xor    dword [counter], 1

; 原子 INC/DEC
lock inc    dword [counter]
lock dec    dword [counter]
```

### LOCK 的实现机制

```
单处理器：  禁止中断即可
多处理器（旧）：总线锁 — 锁住整个内存总线，其他处理器无法访问
多处理器（新）：缓存锁 — 锁定缓存行（cache line），更高效
  - 如果数据在缓存中且不跨缓存行
  - 用 MESI 协议保证一致性
  - 其他核心的缓存行被 invalidate
```

## CMPXCHG — 比较并交换（CAS）

```nasm
; CMPXCHG dst, src
; if (EAX == dst) { dst = src; ZF = 1 }
; else { EAX = dst; ZF = 0 }

; 原子 CAS
lock cmpxchg [shared_var], ebx
jz      .success           ; 交换成功

; 原子条件更新
; if (*ptr == old_val) *ptr = new_val; return success;
atomic_compare_exchange:
    ; 输入：edi = ptr, esi = old_val, edx = new_val
    mov     eax, esi           ; EAX = 期望的旧值
    lock cmpxchg [edi], edx
    setz    al                 ; AL = ZF（是否成功）
    movzx   eax, al
    ret

; CAS 自旋实现原子加
atomic_add:
    ; edi = ptr, esi = delta
    mov     eax, [edi]         ; 读取当前值
.retry:
    mov     edx, eax
    add     edx, esi           ; 计算新值
    lock cmpxchg [edi], edx    ; CAS
    jnz     .retry             ; 失败则重试
    ret
```

## CMPXCHG8B / CMPXCHG16B

```nasm
; CMPXCHG8B — 64 位 CAS（x86-32）
; EDX:EAX = 期望值, ECX:EBX = 新值
; if (EDX:EAX == [mem64]) { [mem64] = ECX:EBX; ZF = 1 }
; else { EDX:EAX = [mem64]; ZF = 0 }

lock cmpxchg8b [qword_var]
jz      .success

; CMPXCHG16B — 128 位 CAS（x86-64）
; RDX:RAX = 期望值, RCX:RBX = 新值
lock cmpxchg16b [oword_var]
jz      .success
```

### 无锁队列示例

```c
// 使用 CMPXCHG8B 实现无锁入队（简化版）
typedef struct { uint32_t low; uint32_t high; } uint64_pair;

int lockfree_push(queue_t *q, void *item) {
    uint64_pair old_val, new_val;
    
    do {
        old_val = *(uint64_pair*)&q->head;
        new_val.low = old_val.low + 1;
        new_val.high = (uint32_t)item;
        
        // CAS8B: 原子更新 head 和 item 指针
    } while (!__sync_bool_compare_and_swap_8(
        (uint64_t*)&q->head,
        *(uint64_t*)&old_val,
        *(uint64_t*)&new_val
    ));
    return 1;
}
```

## XADD — 原子加并返回旧值

```nasm
; XADD dst, reg
; temp = dst; dst += reg; reg = temp

lock xadd   dword [counter], eax    ; [counter] += eax, eax = 旧值

; 原子 fetch-and-add
atomic_fetch_add:
    ; 输入：edi = ptr, eax = delta
    lock xadd [edi], eax
    ; eax = 原值，[edi] = 原值 + delta
    ret

; GCC 内置函数对应
// __sync_fetch_and_add(&counter, 1) → lock xadd [counter], 1
```

## 原子操作的 GCC 内置函数

```c
// GCC/Clang 原子内置函数（编译为对应指令）
int old = __sync_fetch_and_add(&counter, 1);     // XADD
int old = __sync_add_and_fetch(&counter, 1);      // XADD + ADD
int old = __sync_lock_test_and_set(&flag, 1);     // XCHG
int ok = __sync_bool_compare_and_swap(&val, old, new); // CMPXCHG
int old = __sync_val_compare_and_swap(&val, old, new); // CMPXCHG

// C11 _Atomic
#include <stdatomic.h>
atomic_int counter = ATOMIC_VAR_INIT(0);
atomic_fetch_add(&counter, 1);

// C++11 std::atomic
std::atomic<int> counter{0};
counter.fetch_add(1, std::memory_order_relaxed);
```

## 关键要点

> **XCHG 天生原子**：不需要 LOCK 前缀。自旋锁的基础。

> **LOCK 前缀**使普通算术/逻辑指令变为原子：`LOCK ADD`, `LOCK SUB` 等。现代 CPU 用缓存锁而非总线锁，开销很小。

> **CMPXCHG = CAS**：无锁编程的核心原语。配合自旋重试实现任意原子操作。

> **CMPXCHG8B/16B**：双字/四字 CAS，实现 64/128 位无锁操作。C++ `std::atomic<double>` 在 x86-64 上编译为 `CMPXCHG8B` 或 `CMPXCHG16B`。
