IEEE 754 标准定义了浮点数的二进制表示格式——用**符号位 + 指数 + 尾数**编码实数，是几乎所有现代 CPU 和编程语言的浮点运算基础。

## IEEE 754 格式

### float（单精度，32 位）

```
 31  30──────23  22──────────────0
┌───┬──────────┬──────────────────┐
│ S │ 指数(8位) │   尾数(23位)     │
└───┴──────────┴──────────────────┘

值 = (-1)^S × 2^(指数-127) × 1.尾数
```

### double（双精度，64 位）

```
 63  62────────52  51──────────────────0
┌───┬────────────┬──────────────────────┐
│ S │ 指数(11位)  │     尾数(52位)       │
└───┴────────────┴──────────────────────┘

值 = (-1)^S × 2^(指数-1023) × 1.尾数
```

### 示例：3.14 的 IEEE 754 编码

```
3.14 (float)：

1. 整数部分：3 = 11₂
2. 小数部分：0.14 ≈ .00100011110101110000101₂
3. 规格化：1.10010001111010111000010 × 2^1
4. 指数：1 + 127 = 128 = 10000000₂
5. 尾数：10010001111010111000010

最终：0 10000000 10010001111010111000010
     = 0x4048F5C3
```

## 特殊值

```
float 类型       符号  指数        尾数          值
─────────────────────────────────────────────────
+0.0             0    00000000   00000000...    +0
-0.0             1    00000000   00000000...    -0
正无穷 +∞       0    11111111   00000000...    ∞
负无穷 -∞       1    11111111   00000000...   -∞
NaN (Quiet)      x    11111111   1xxxxxxx...    NaN
NaN (Signal)     x    11111111   0xxxxxxx...    sNaN

非规格化数       x    00000000   非零尾数       ±0.尾数 × 2^(-126)
```

```nasm
    pos_inf:  dd 0x7F800000        ; +∞
    neg_inf:  dd 0xFF800000        ; -∞

```

## C 语言中的浮点操作

```c
#include <math.h>
#include <float.h>
#include <stdio.h>

int main(void) {
    // 特殊值检查
    double x = 1.0 / 0.0;        // +∞
    printf("isinf: %d\n", isinf(x));        // 1
    
    double y = 0.0 / 0.0;        // NaN
    printf("isnan: %d\n", isnan(y));        // 1
    
    // ❌ 永远不要直接比较浮点数！
    double a = 0.1 + 0.2;
    if (a == 0.3) {              // false！
        printf("相等\n");        // 不会执行
    }
    printf("实际值: %.20f\n", a); // 0.30000000000000004441
    
    // ✅ 使用 epsilon 比较
    double epsilon = 1e-9;
    if (fabs(a - 0.3) < epsilon) {
        printf("近似相等\n");
    }
    
    // 浮点数精度极限
    printf("FLT_EPSILON:  %e\n", FLT_EPSILON);   // 1.19209e-07
    printf("DBL_EPSILON:  %e\n", DBL_EPSILON);   // 2.22045e-16
    
    return 0;
}
```

## 浮点运算的陷阱

```c
// 陷阱 1: 精度丢失
float f = 16777216.0f;           // 2^24
f += 1.0f;
printf("%.1f\n", f);             // 16777216.0 — 加 1 没效果！
// 原因：float 只有 23 位尾数，精度约 7 位十进制

// 陷阱 2: 顺序影响结果
float sum = 0.0f;
for (int i = 0; i < 1000000; i++)
    sum += 1.0f;
printf("%.1f\n", sum);           // 不一定是 1000000.0

// 更安全的方式：Kahan 求和
float kahan_sum(float *arr, int n) {
    float sum = 0.0f, c = 0.0f;
    for (int i = 0; i < n; i++) {
        float y = arr[i] - c;
        float t = sum + y;
        c = (t - sum) - y;       // 丢失的低位
        sum = t;
    }
    return sum;
}

// 陷阱 3: NaN 传播
double x = 0.0 / 0.0;           // NaN
double y = x + 1.0;             // NaN（任何运算含 NaN = NaN）
double z = x * 0.0;             // NaN（不是 0！）

// 陷阱 4: -0.0 的特殊行为
double neg_zero = -0.0;
printf("1/neg_zero = %f\n", 1.0/neg_zero);  // -∞
printf("+0 == -0: %d\n", +0.0 == -0.0);      // true（相等比较）
printf("atan2(-0,-1) = %f\n", atan2(-0.0, -1.0)); // -π
```

## 浮点数在汇编中的使用

```nasm
; SSE 中的浮点运算
section .data
    align 16
    a: dd 1.0, 2.0, 3.0, 4.0     ; 4 个 float
    b: dd 5.0, 6.0, 7.0, 8.0

section .text
global float_ops
float_ops:
    movaps  xmm0, [a]
    movaps  xmm1, [b]
    addps   xmm0, xmm1            ; 打包加法
    mulps   xmm0, xmm1            ; 打包乘法
    ; 结果在 xmm0
    ret

; x87 中的浮点运算
global fpu_add
fpu_add:
    fld     dword [esp+4]         ; 加载第一个 float
    fadd    dword [esp+8]         ; 加上第二个
    ; 结果在 ST(0)
    ret
```

## 类型转换

```c
// float → int 的截断行为
float f = 2.999f;
int i = (int)f;                   // 2 — 向零截断
int j = (int)roundf(f);           // 3 — 四舍五入
int k = (int)ceilf(f);            // 3 — 向上取整
int l = (int)floorf(f);           // 2 — 向下取整

// int → float 可能丢失精度
int big = 16777217;               // 2^24 + 1
float f2 = (float)big;
printf("%d vs %.0f\n", big, f2);  // 16777217 vs 16777216 — 精度丢失！
```

## 关键要点

> **IEEE 754 = 符号位 + 指数 + 尾数**：float (1+8+23), double (1+11+52)。指数用偏移码（127/1023）。

> **永远不要直接比较浮点数**：用 `fabs(a-b) < epsilon`。0.1 + 0.2 ≠ 0.3。

> **特殊值**：±0, ±∞, NaN 有特定用途。NaN 不等于任何值（包括自己）。`isnan()` 而不是 `==` 检测。

> **float 精度约 7 位十进制，double 约 15 位**。超过这个精度的数字会丢失低位。大整数转 float 会丢失精度。
