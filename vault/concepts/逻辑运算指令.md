逻辑运算指令对操作数进行按位逻辑操作，是位操作、掩码提取和条件判断的基础工具。

## AND — 按位与

```asm
and eax, ebx       ; EAX &= EBX
and eax, 0xFF      ; 只保留低 8 位（提取字节）
and eax, 0xFFFFFFF0  ; 低 4 位清零（对齐到 16 字节）

; 常见用法：掩码提取
and eax, 0b00001111  ; 提取低 4 位

; AND 影响标志：ZF, SF, PF（CF=OF=0）
```

## OR — 按位或

```asm
or eax, ebx        ; EAX |= EBX
or eax, 0x01       ; 设置最低位
or eax, 0x80000000 ; 设置最高位

; 常见用法：设置特定位
or eax, 0b00000100  ; 设置第 2 位

; OR 影响标志：ZF, SF, PF（CF=OF=0）
```

## XOR — 按位异或

```asm
xor eax, ebx       ; EAX ^= EBX
xor eax, eax       ; EAX = 0（清零，2 字节，比 MOV eax,0 短）
xor eax, 0xFF      ; 翻转低 8 位

; XOR 影响标志：ZF, SF, PF（CF=OF=0）
```

### XOR 清零 — 汇编中最常见的指令之一

```asm
xor eax, eax       ; EAX = 0（2 字节：31 C0）
; vs
mov eax, 0         ; EAX = 0（5 字节：B8 00 00 00 00）

; 编译器总是选择 xor reg, reg
xor ecx, ecx       ; 循环计数器清零
xor edx, edx       ; 准备 DIV 的高位
```

## NOT — 按位取反

```asm
not eax            ; EAX = ~EAX（每一位取反）

; NOT 不影响任何标志位！
; 注意区别：NOT 是按位取反，NEG 是求负
; NOT 0xFF → 0x00,  NEG 0xFF → 0x01
```

## TEST — 测试（AND 但不保存结果）

```asm
test eax, ebx      ; EAX & EBX，只设标志，不修改 EAX
test eax, eax      ; 检查 EAX 是否为 0（比 CMP eax, 0 短 1 字节）
test al, 0x01      ; 检查最低位是否为 1

; TEST 影响标志：ZF, SF, PF（CF=OF=0）

; 常见模式：检测特定位
test eax, 0x80     ; 检查第 7 位
jnz bit_is_set     ; ZF=0 表示位为 1

; 检查偶数
test eax, 1
jz is_even         ; 最低位为 0 → 偶数
```

## 常见技巧集

### 1. 清零

```asm
xor eax, eax       ; 最优清零（2 字节，且打破依赖链）
xor ecx, ecx
xor edx, edx
```

### 2. 位掩码提取

```asm
; 提取低 4 位
and eax, 0x0F

; 提取第 8-15 位
shr eax, 8
and eax, 0xFF

; 提取颜色分量（RGBA）
mov eax, [pixel]
mov ebx, eax
and eax, 0x00FF0000  ; 提取 R 分量
shr eax, 16           ; 右移到低 8 位
```

### 3. 位设置/清除/翻转

```asm
; 设置第 3 位
or eax, 0x08

; 清除第 3 位
and eax, ~0x08

; 翻转第 3 位
xor eax, 0x08
```

### 4. 取绝对值（无分支）

```asm
; 取 EAX 的绝对值
mov ebx, eax
sar ebx, 31        ; EBX = 符号掩码（全 0 或全 1）
xor eax, ebx
sub eax, ebx       ; EAX = |EAX|

; 等价于：
; if (EAX < 0) EAX = -EAX;
```

### 5. 交换（无临时变量）

```asm
; 交换 EAX 和 EBX（无临时变量）
xor eax, ebx
xor ebx, eax
xor eax, ebx       ; EAX 和 EBX 已交换
```

### 6. 判断奇偶

```asm
; 判断 EAX 是否为偶数
test eax, 1
jz is_even         ; 最低位为 0 → 偶数
; 等价于用 AND: and eax, 1; jz is_even
```

### 7. 对齐地址

```asm
; 向下对齐到 16 字节边界
and eax, -16       ; -16 = 0xFFFFFFF0

; 向上对齐到 16 字节
add eax, 15
and eax, -16

; 向下对齐到 4KB
and eax, -4096     ; -4096 = 0xFFFFF000
```

> [!tip] 关键要点
> - XOR reg, reg 是最高效的清零方式（编译器首选）
> - AND 用于掩码提取和对齐操作
> - TEST = AND 不保存结果，只设标志（检测特定位）
> - NOT 不影响标志位，NEG 影响标志位（注意区别）
> - 位操作技巧在底层编程中无处不在：颜色处理、标志管理、对齐
