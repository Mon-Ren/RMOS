x86 提供 DR0-DR7 八个调试寄存器，支持硬件断点（无需修改代码）和单步执行，是调试器（GDB、OllyDbg）和反调试技术的硬件基础。

## 调试寄存器概览

```
DR0 — 断点地址 0
DR1 — 断点地址 1
DR2 — 断点地址 2
DR3 — 断点地址 3
DR4 — 保留（旧 CPU）
DR5 — 保留（旧 CPU）
DR6 — 断点状态（哪些断点触发）
DR7 — 断点控制（启用、条件、长度）
```

## DR0-DR3 — 断点地址

```nasm
; 设置硬件断点
mov     eax, breakpoint_addr
mov     dr0, eax               ; 断点 0 地址
mov     dr1, eax               ; 断点 1 地址（最多 4 个）
mov     dr2, eax
mov     dr3, eax
```

每个 DRn 存储一个线性地址。当 CPU 访问（读/写/执行）这个地址时触发断点异常。

## DR7 — 断点控制

```
DR7 格式：
  位 0: L0 — 局部断点 0 启用（当前任务）
  位 1: G0 — 全局断点 0 启用（所有任务）
  位 2: L1 — 局部断点 1 启用
  位 3: G1 — 全局断点 1 启用
  ...
  位 7: G3 — 全局断点 3 启用

  位 16-17: R/W0 — 断点 0 的条件
  位 18-19: LEN0 — 断点 0 的长度
  位 20-21: R/W1
  位 22-23: LEN1
  ...

R/W 编码：
  00 = 仅执行断点
  01 = 写入断点
  10 = I/O 断点（需 CR4.DE=1）
  11 = 读写断点

LEN 编码：
  00 = 1 字节
  01 = 2 字节
  10 = 8 字节（x86-64）/ 保留（x86-32）
  11 = 4 字节
```

```nasm
; 设置 DR7 示例
; 断点 0: 执行断点, 1 字节
; 断点 1: 写入断点, 4 字节
mov     eax, 0x000A0001
;  bit 0 (L0) = 1          启用断点 0
;  bit 16-17 (R/W0) = 00   仅执行
;  bit 18-19 (LEN0) = 00   1 字节
;  bit 2 (L1) = 0, bit 3 (G1) = 1  启用断点 1
;  bit 20-21 (R/W1) = 01   写入断点
;  bit 22-23 (LEN1) = 10   4 字节
mov     dr7, eax
```

### 设置断点的完整示例

```c
// 设置硬件断点（Ring 0）
void set_hw_breakpoint(int reg, void *addr, int type, int len) {
    uint64_t dr7;
    
    // 设置地址寄存器
    switch (reg) {
        case 0: __asm__ __volatile__ ("mov %0, %%dr0" : : "r"(addr)); break;
        case 1: __asm__ __volatile__ ("mov %0, %%dr1" : : "r"(addr)); break;
        case 2: __asm__ __volatile__ ("mov %0, %%dr2" : : "r"(addr)); break;
        case 3: __asm__ __volatile__ ("mov %0, %%dr3" : : "r"(addr)); break;
    }
    
    // 读取 DR7
    __asm__ __volatile__ ("mov %%dr7, %0" : "=r"(dr7));
    
    // 清除该断点的旧设置
    dr7 &= ~(0xF << (reg * 4 + 16));  // 清除 R/W 和 LEN
    dr7 |= (3 << (reg * 2));           // 设置 Ln 和 Gn（局部+全局启用）
    dr7 |= (type << (reg * 4 + 16));   // 设置 R/W
    dr7 |= (len  << (reg * 4 + 18));   // 设置 LEN
    
    // 写入 DR7
    __asm__ __volatile__ ("mov %0, %%dr7" : : "r"(dr7));
}

// 使用
set_hw_breakpoint(0, &my_variable, 1, 3);  // 当写入 my_variable 时断点
set_hw_breakpoint(1, my_function, 0, 0);   // 当执行 my_function 时断点
```

## DR6 — 断点状态

```
DR6 格式：
  位 0: B0 — 断点 0 被触发
  位 1: B1 — 断点 1 被触发
  位 2: B2 — 断点 2 被触发
  位 3: B3 — 断点 3 被触发
  位 13: BD — 调制寄存器访问断点
  位 14: BS — 单步断点
  位 15: BT — 任务切换断点
```

```nasm
; Debug Exception (#DB) 处理程序
debug_handler:
    mov     eax, dr6               ; 读取状态
    test    eax, 0x0F              ; 哪个断点触发了？
    jnz     .hw_breakpoint
    test    eax, (1 << 14)         ; BS 位 = 单步断点
    jnz     .single_step
    test    eax, (1 << 15)         ; BT 位 = 任务切换
    jnz     .task_switch
    iret

.hw_breakpoint:
    ; 检查每个 B0-B3 位确定哪个断点
    test    eax, 1
    jnz     .bp0_triggered
    ; ...
    iret

.single_step:
    ; 单步执行的处理
    iret
```

## INT3 软断点 vs DR 硬断点

```nasm
; === INT3 软断点 ===
; 调试器将目标地址的第一个字节替换为 0xCC (INT3)
; 触发 Breakpoint Exception (#BP, 异常号 3)

; 原始代码：0x1000: B8 01 00 00 00  (mov eax, 1)
; 设断点后：0x1000: CC 00 00 00 00  (INT3 + 垃圾)

; 调试器流程：
; 1. 读取 0x1000 的字节 → 保存
; 2. 写入 0xCC → INT3
; 3. 用户运行到 0x1000 → 触发 #BP
; 4. 调试器恢复原字节，显示断点状态
; 5. 继续执行时，单步越过这条指令

; === 对比 ===
; INT3:  修改代码，数量无限，有副作用（修改内存）
; 硬断点: 不修改代码，最多 4 个，无副作用
```

## 单步执行：EFLAGS.TF

```nasm
; 设置 TF 标志开启单步执行
pushf                       ; EFLAGS 入栈
or      dword [esp], 0x100  ; 设置 TF (bit 8)
popf                        ; 加载修改后的 EFLAGS

; 执行完下一条指令后，CPU 自动触发 #DB 异常
; Debug Exception Handler 中清除 TF 以停止单步

; 调试器中常用
; 1. 设置 EFLAGS.TF = 1
; 2. 执行一条指令 → #DB
; 3. 在 #DB handler 中记录状态
; 4. 重复 2-3 直到断点或完成
```

## 反调试技术

```c
// 检测硬件断点 — 检查 DR0-DR3 是否非零
int detect_hw_breakpoints(void) {
    uint64_t dr0, dr1, dr2, dr3;
    __asm__ __volatile__ ("mov %%dr0, %0" : "=r"(dr0));
    __asm__ __volatile__ ("mov %%dr1, %0" : "=r"(dr1));
    __asm__ __volatile__ ("mov %%dr2, %0" : "=r"(dr2));
    __asm__ __volatile__ ("mov %%dr3, %0" : "=r"(dr3));
    return (dr0 | dr1 | dr2 | dr3) != 0;
}

// 清除硬件断点
void clear_hw_breakpoints(void) {
    __asm__ __volatile__ (
        "xor %%rax, %%rax\n\t"
        "mov %%rax, %%dr0\n\t"
        "mov %%rax, %%dr1\n\t"
        "mov %%rax, %%dr2\n\t"
        "mov %%rax, %%dr3\n\t"
        "mov %%rax, %%dr7\n\t"
        ::: "rax"
    );
}
```

## 关键要点

> **4 个硬件断点**：DR0-DR3 各存储一个地址，DR7 控制启用和条件。比软断点多开销但不改代码。

> **三种断点类型**：执行（00）、写入（01）、读写（11）。I/O 断点需要 CR4.DE=1。

> **DR6 记录触发状态**：中断处理程序读 DR6 判断哪个断点触发。调试器必须在退出前清零 DR6。

> **单步 = EFLAGS.TF**：置 1 后每执行一条指令触发 #DB。GDB 的 `stepi` 就是用这个。
